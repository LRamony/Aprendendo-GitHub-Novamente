#Crie um novo repositório na linha de comando:
git init
git add README.md
git commit -m "first commit"
git branch -M main
git remote add origin https://github.com/LRamony/Aprendendo-GitHub-Novamente.git
git push -u origin main

#Ou envie um repositório existente a partir da linha de comando:

git remote add origin https://github.com/LRamony/Aprendendo-GitHub-Novamente.git
git branch -M main
git push -u origin main

#git branch < mostra em qual branch vc esta trabalhando
#git checkout <Nome da branch> < Muda para a branch desejada
#git checkout -b <Nome da branch> < Cria e seta para a nova branch
#git reflog < serve para ver o historico das atualizações
#git reset --hard <ID pego com o reflog> < Como navegar entre versões do seu código
#git merge <Nome na branch> < Como unir as branch ou código
#git pull < serve para atualizar o codigo na sua maquina com o codigo no github
#touch .gitignore < So salvar dentro desse arquivo o q deseja q seja ignorado
